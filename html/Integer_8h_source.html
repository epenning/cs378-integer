80 template <typename II1, typename II2, typename FI>
81 FI plus_digits (II1 b1, II1 e1, II2 b2, II2 e2, FI x) {
82  int carry = 0;
83  int plus1 = 0;
84  int plus2 = 0;
85  int num = 0;
86  while (b1 != e1 || b2 != e2) {
87  if (b1 == e1) {
88  // b1 done, rest are from b2
89  plus1 = 0;
90  plus2 = *b2;
91  ++b2;
92  }
93  else if (b2 == e2) {
94  // b2 done, rest are from b1
95  plus1 = *b1;
96  plus2 = 0;
97  ++b1;
98  }
99  else {
100  // both have digits
101  plus1 = *b1;
102  plus2 = *b2;
103  ++b1;
104  ++b2;
105  }
106  num = plus1 + plus2 + carry;
107  carry = num / 10;
108  *x = num % 10;
109  ++x;
110  }
111  if (carry) {
112  *x = carry;
113  ++x;
114  }
115  return x;
116 }

134 template <typename II1, typename II2, typename FI>
135 FI minus_digits (II1 b1, II1 e1, II2 b2, II2 e2, FI x) {
136  FI originalX = x;
137  int zeroCount = 0;
138  int borrow = 0;
139  int sub1 = 0;
140  int sub2 = 0;
141  bool empty = true;
142  // b2 guaranteed to reach end before or same time as b1
143  while (b1 != e1) {
144  if (b2 == e2) {
145  // b2 done, rest are from b1
146  sub1 = *b1;
147  sub2 = 0;
148  ++b1;
149  if (borrow) {
150  sub1 -= borrow;
151  borrow = 0;
152  }
153  if (sub1 == 0 && b1 == e1) {
154  break;
155  }
156  }
157  else {
158  // both have digits
159  sub1 = *b1;
160  sub2 = *b2;
161  ++b1;
162  ++b2;
163  if (borrow) {
164  sub1 -= borrow;
165  borrow = 0;
166  }
167  }
168  if (sub1 < sub2) {
169  borrow++;
170  sub1 += 10;
171  }
172  if ((sub1-sub2) != 0 || !empty) {
173  if (b1 == e1 && sub1-sub2 == 0) {
174  break;
175  }
176  *x = sub1 - sub2;
177  ++x;
178  empty = false;
179  }
180  else {
181  zeroCount++;
182  }
183  }
184  if (empty) {
185  // diff was 0
186  *x = 0;
187  ++x;
188  }
189  else if (zeroCount){
190  x = shift_right_digits(originalX, x, zeroCount, originalX);
191  }
192  return x;
193 }

210 template <typename II1, typename II2, typename FI>
211 FI multiplies_digits (II1 b1, II1 e1, II2 b2, II2 e2, FI x) {
212  vector<int> vec1(b1, e1);
213  vector<int> vec2(b2, e2);
214 
215  // base case: multiply one digit by other number
216  if(vec1.size() < 2 || vec2.size() < 2){
217  vector<int> product(0);
218  vector<int>::iterator productEnd;
219  if(vec1.size() < vec2.size()){
220  if(vec1.size() == 0){
221  product = vec2;
222  productEnd = product.end();
223  }else{
224  product.resize(vec2.size() + 1);
225  productEnd = product.begin() + 1;
226  // add other number to product, digit times
227  for(int i = 0; i < vec1[0]; ++i){
228  productEnd = plus_digits(product.begin(), productEnd, vec2.begin(), vec2.end(), product.begin());
229  }
230  }
231  }else{
232  if(vec2.size() == 0){
233  product = vec1;
234  productEnd = product.end();
235  }else{
236  product.resize(vec1.size() + 1,0);
237  productEnd = product.begin() + 1;
238  // add other number to product, digit times
239  for(int i = 0; i < vec2[0]; ++i){
240  productEnd = plus_digits(product.begin(), productEnd, vec1.begin(), vec1.end(), product.begin());
241  }
242  }
243  }
244  // output product
245  vector<int>::iterator i = product.begin();
246  while(i != productEnd){
247  *x = *i;
248  ++x;
249  ++i;
250  }
251  return x;
252  }
253  // Karatsuba algorithm
254  int m = max(vec1.size(), vec2.size());
255  int m2 = ceil(m/2.0);
256  if(min(vec1.size(), vec2.size()) <= m2){
257  m2 = min(vec1.size(), vec2.size()) - 1;
258  }
259  // set up halfway-point iterators
260  vector<int>::iterator low1B = vec1.begin();
261  vector<int>::iterator low1E = vec1.begin() + m2;
262  vector<int>::iterator high1B = vec1.begin() + m2;
263  vector<int>::iterator high1E = vec1.end();
264  vector<int>::iterator low2B = vec2.begin();
265  vector<int>::iterator low2E = vec2.begin() + m2;
266  vector<int>::iterator high2B = vec2.begin() + m2;
267  vector<int>::iterator high2E = vec2.end();
268  // compute z0
269  vector<int> z0(2*m);
270  vector<int>::iterator z0e = multiplies_digits(low1B, low1E, low2B, low2E, z0.begin());
271  // compute z1
272  vector<int> z1(2*m+1);
273  vector<int> multi1(m+2);
274  vector<int> multi2(m+2);
275  vector<int>::iterator one = plus_digits (low1B, low1E, high1B, high1E, multi1.begin());
276  vector<int>::iterator two = plus_digits (low2B, low2E, high2B, high2E, multi2.begin());
277  vector<int>::iterator z1e = multiplies_digits(multi1.begin(), one, multi2.begin(), two, z1.begin());
278  // compute z2
279  vector<int> z2(2*m);
280  vector<int>::iterator z2e = multiplies_digits(high1B, high1E, high2B, high2E, z2.begin());
281  // compute z2*10^(2*m2)
282  vector<int> partOne(2*m);
283  vector<int>::iterator partOneEnd = shift_right_digits (z2.begin(), z2e, 2*m2, partOne.begin());
284  // compute (z1 - z2 - z0)*10^(m2)
285  vector<int> subOne(2*m);
286  vector<int>::iterator subOneEnd = minus_digits (z1.begin(), z1e, z2.begin(), z2e, subOne.begin());
287  vector<int> subTwo(2*m);
288  vector<int>::iterator subTwoEnd = minus_digits (subOne.begin(), subOneEnd, z0.begin(), z0e, subTwo.begin());
289  vector<int> partTwo(subTwo.size() + m2);
290  vector<int>::iterator partTwoEnd = shift_right_digits (subTwo.begin(), subTwoEnd, m2, partTwo.begin());
291  // compute last two results added together + z2
292  vector<int> addOne(partOne.size());
293  vector<int>::iterator addOneEnd = plus_digits(partOne.begin(), partOneEnd, partTwo.begin(), partTwoEnd, addOne.begin());
294  return plus_digits(addOne.begin(), addOneEnd, z0.begin(), z0e, x);
295 }

312 template <typename II1, typename II2, typename FI>
313 FI divides_digits (II1 b1, II1 e1, II2 b2, II2 e2, FI x) {
314  //simpile solution is to use subtraction
315  vector<int> numerator(b1, e1);
316  vector<int> denominator(b2, e2);
317  vector<int> quotient(numerator.size());
318  vector<int> one(1, 1);
319  vector<int>::iterator quotientEnd = quotient.begin() + 1;
320  vector<int>::iterator numEnd = numerator.end();
321 
322  while (true) {
323  // count number of times denominator can be subtracted from numerator
324  if((numEnd - numerator.begin()) < denominator.size()){
325  break;
326  }
327  else if ((numEnd - numerator.begin()) == denominator.size()){
328  for(int i = (numEnd - numerator.begin()) - 1; i >= 0; --i){
329  if (numerator[i] > denominator[i])
330  break;
331  else if((numerator[i] < denominator[i])){
332  goto finish;
333  }
334  }
335  }
336  numEnd = minus_digits(numerator.begin(), numEnd, denominator.begin(), denominator.end(), numerator.begin());
337  quotientEnd = plus_digits(quotient.begin(), quotientEnd, one.begin(), one.end(), quotient.begin());
338  }
339  finish:
340  return copy(quotient.begin(), quotientEnd, x);
341 }
